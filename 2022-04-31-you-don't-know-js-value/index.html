배열

자바스크립트의 배열은 다른 언어와 달리 어떤 타입의 값이라도 담을 수 있는 그릇이다.

var a = [1, '2', [3]]

a.length // 3
a[0] === 1 //true
a[2][0] === 3 //true

배열 크기도 미리 정하지 않고 선언할 수 잇으며 원하는 값을 추가하면 된다.

var a = []
a.length // 0
a[0] = 1
a[1] = '2'
a[2] = [3]

a.length // 3

빈 슬롯이 있는 배열은 혼란을 부축 일 수 있다.

var a = []
a[0] = 1
//a[1]을 건너뛰자
a[2] = [3]

a[1] //undefined

a.length //3

명시적으로 a[1] = undefined로 세팅한 것과 세팅을 안한 빈 a[1]은 다르다.

배열 인덱스는 숫자이지만, 배열 자체가 하나의 객체여서 키/프로퍼티 문자열을 추가할 수 있다. (하지만 length는 증가하지 않는다.)

var a = []
a[0] = 1
a['foobar'] = 2

a.length //1
a['foobar'] //2
a.foobar //2

그런데 키로 넣은 문자열 값이 표준 10진수 숫자로 타입이 바뀌면, 마치 문자열 키가 아닌 숫자 키를 사용한 것과 같은 결과가 초래되므로 주의할 필요가 있따.

var a = []
a['13'] = 42
a.length //14

유사 배열 값을 진짜 배열로 바꾸고 싶을 떄는 배열 유틸리티 함수를 사용하여 해결하는 게 일반적이다.

arguments 객체에서 인자를 리스트를 가져오는 변환을 살펴보자.

function foo() {
  var arr = Array.prototype.slice.call(arguments)
  arr.push('bam')
  console.log(arr)
}
foo('bar', 'baz') //["bar", "baz", "bam"]

ES6부터는 Array.from()을 사용하면 된다.

var arr = Array.from(arguments)

문자열

흔히 다른 언어들에서 문자열은 문자의 배열이지만 자바스크립트에서는 생김새만 비슷할 뿐 문자 배열과 같지 않다.

var a = 'foo'
var b = ['f', 'o', 'o']

문자열은 배열과 겉모습이 닮았다.(유사배열임)

양쪽 모두 length 프로퍼티, indexOf() 메서드, concat () 메서드를 가진다.

a.length // 3
b.length // 3

a.indexOf('o') // 1
b.indexOf('o') // 1

var c = a.concat('bar') // "foobar"
var d = b.concat(['b', 'a', 'r']) // ["f", "o", "o", "b", "a", "r"];
a === c //false
b === d //false

a // "foo"
b //["f","o","o"]

문자열은 불변 값이고 배열은 가변 값이다.

a[1] = 'O'
b[1] = 'O'

a // "foo" 불변
b // ["f","O","o"] 가변

심지어 IE 구버전은 문자열을 배열처럼 접근하는 것을 문법에러로 인식한다.

문자열을 다룰 때 유용한 대부분의 배열 메서드는 사실상 문자열에 쓸 수 없지만 문자열에 대해 불변 배열 메서드를 빌려 쓸 수 있다.

c = a.toUpperCase()
a == c //false
a //"foo"
c //"FOO"

b.push('!')
b //["f", "O" "o", "!"]
a.join //undefined
a.map //undefined

var c = Array.prototype.join.call(a, '-')
var d = Array.prototype.map.call(a, v => v.toUppercase() + '.').join('')

c //"f-o-o"
d //"F.O.O."

문자열에는 배열의 가변 메서드인 reverse등을 빌려쓸 수가 없기 때문에

var c = a.split('').reverse().join('')

문자열 → 배열(split) → 뒤집기(reverse) → 문자열로 만들기(join)

문자열 자체에 많은 작업이 필요한 경우에는 배열로 관리를 한다음 join으로 합치는 게 나을 수 있다.

숫자

자바스크립트의 숫자 타입은 number가 유일하며

정수는 부동 소수점 값이 없는 값이다. (42===42.0)

숫자 값은 Number 객체 래퍼로 박싱할 수 있기 때문에 Number.prototype 메서드로 접근할 수 있다.

var a = 42.59
a.toFixed(0) //"43"
a.toFixed(1) //"42.6"
a.toFixed(2) //"42.59"
a.toFixed(3) //"42.590"

작은 소수 값

0.1 + 0.2 === 0.3 //false

이것이 자바스크립트의 이진 부동 소수점 숫자의 부작용이다. 자바스크립트의 0.1은 정확히 0.1이 아니고 0.2도 정확히 0.2가 아니므로 실제로 0.1 + 0.2는 0.3000000004에 가깝다.

그렇다면 이를 어떻게 비교해야할까?

머신 입실론을 사용할 수 있다.

function numbersCloseEnoughToEqual(n1, n2) {
  return Math.abs(n1 - n2) < Number.EPSILON
}

var a = 0.1 + 0.2
var b = 0.3

numbersCloseEnoughToEqual(a, b) //true
numbersCloseEnoughToEqual(0.0000001, 0.0000002) //false

특수 값

값 아닌 값

undefined 타입의 값은 undefined밖에 없다. null 타입의 값은 null뿐이다. 그래서 이 둘은 타입과 값이 항상 값다.

void 연산자

var a = 42
console.log(void a, a) //undefined 42

void 연산자는 어떤 표현식의 결괏값이 없다는 걸 확실히 밝혀야할 때 요긴하다.

특수 문자

NaN은 숫자 아님보다 몹쓸 숫자라고 하는 게 더 정확하다.

var a = 2 / foo //NaN
typeof a === 'number' //true

그리고 NaN은 비교 불능이다.

var a = 2 / 'foo'
a == NaN // false
a === NaN // false

그럼 NaN여부는 어떻게 확인하는가?

내장 전역 함수인 isNaN도 가능하겠지만 문자열등을 isNaN했을 때도 true를 반환하므로 이런 문제를 해결한 게 Number 객체에 있는 Number.isNaN이다. ES6버전 이하에서는 아래 코드로 안전하게 NaN 여부를 체크할 수 있다.

if (!Number.isNaN) {
  Number.isNaN = function (n) {
    return typeof n === 'number' && window.isNaN(n)
  }
}
var a = 2 / 'foo'
var b = 'foo'

Number.isNaN(a) //true
Number.isNaN(b) //false ...휴, 다행이다!

무한대

자바스크립트에서는 0으로 나누기 연산이 정의되어있으므로

var a = 1 / 0 //Infinity
var b = -1 / 0 //-Infinity

자바스크립트는 유한 숫자 표현식을 사용하므로 수학 교과서와는 다르게 덧셈, 뺄셈 같은 연산 결과가 +무한대/-무한대가 될 수 있다.

var a = Number.MAX_VALUE //1.7976931348623157e+308
a + a //Infinity
a + Math.pow(2, 970) //Infinity
a + Math.pos(2, 969) //1.7976931348623157e+308

영(0)

자바스크립트엔 양수 영(0)과 음의 영(-0)이 있다.

표기만 음의 영인 것이 아니라 수식의 연산결과 또한 -0이 되곤한다.

var a = 0 / 3
var b = 0 * -3

-0을 문자열화하면 항상 “0”이다.

var a = 0 / -3
a // -0
// 명세는 거짓말을 하라고 시킨다!
a.toString() // "0"
a + '' //"0"
String(a) //"0"

//Json조차 속아 넘어간다.
JSON.stringify(a) // "0"

//하지만 문자열 ->숫자는 있는 그대로 보여주는걸
;+'-0' // -0
Number('0') // -0
JSON.parse('-0') // -0

비교 연산 역시 믿을 수 없다.

var a = 0
var b = 0 / 3

a == b // true
;-0 == 0 // true

a === b // true
;-0 === 0 // true

0 > -0 // false;
a > b // false

그러면 -0과 0을 구분하려면 어떻게 해야할까?

function isNegZero(n) {
  n = Number(n)
  return n === 0 && 1 / n === -Infinity
}

isNegZero(-0) //true
isNegZero(0 / -3) //true
isNegZero(0) //false

-0은 왜 있는 걸까?

값의 크기로 어떤 정보(예시: 애니메이션 프레임당 넘김 속도)와 그 값의 부호로 또 다른 정보(예시: 넘김 방향)를 동시에 나타내야 하는 상황을 구현해야하기 때문이다.

변수값이 0에 도달하여 부호가 바뀌는 순간 그 직전까지의 이동방향을 알기 위해 부호가 다른 두 0은 유용하다.

잠재적인 정보(방향) 등의 소실을 방지하기 위해 0의 부호를 보존했다.

특이한 동등 비교

ES6부터는 잡다한 예외를 걱정하지 않아도 두 값이 절대적으로 동등한지를 확인하는 새로운 유틸리티를 지원한다.

바로 Object.is()다.

var a = 2 / foo
var b = -3 / 0

Object.is(a, NaN) // true
Object.is(b, -0) // true
Object.is(b, 0) //false

Object.is()를 만들어보자

if (!Object.is) {
  Object.is = function (v1, v2) {
    //'-0'
    if (v1 === 0 && v2 === 0) {
      return 1 / v1 === 1 / v2 //만약 -0 은 1/-0이 음의 무한대이므로
    }
    //'NaN'
    if (v1 !== v1) {
      return v2 !== v2
    }
    // else
    return v1 === v2
  }
}

값 vs 레퍼런스

C++ vs JavaScript

C++에서는 어떤 함수에 전달한 숫자 인자 값을 그 함수 내에서 수정하려면 int &myNum 형태로 함수 인자를 선언하여 주소를 받고 호출하는 쪽에서는 변수 x를 넘기면 myNum은 x를 참조한다. 레퍼런스는 포인터의 특수한 형태로 다른 변수의 포인터를 가진다. 레퍼런스 인자를 선언하지 않으면 전달한 값은 아무리 복잡한 객체일지라도 언제나 복사되는 특성이 있따.

자바스크립트는 포인터라는 개념 자체가 없고 참조하는 방법도 조금 다르다. 변수가 다른 변수를 참조할 수 없다.

자바스크립트에서 레퍼런스는 값이므로 서로 다른 10개의 레퍼런스가 있다면 이들은 저마다 항상 공유된 단일 값을 개별적으로 참조한다. 또 자바스크립트에는 값 또는 레퍼런스의 할당 및 전달을 제어하는 구문 암시가 없다. 오직 값의 타입만으로 값-복사, 레퍼런스-복사 둘 중 한쪽이 결정된다.

var a = 2
var b = a // b는 언제나 a에서 값을 복사한다.
b++
a // 2
b // 3

var c = [1, 2, 3]
var d = c // d는 공유된 [1,2,3]값의 레퍼런스이다.
d.push(4)
c // [1,2,3,4]
d // [1,2,3,4]

null, undefined, string, number, boolean, symbol 같은 스칼라원시값은 언제나 값-복사방식으로 할당/전달된다.

객체나 함수 등 합성 값은 할당/전달시 반드시 레퍼런스 사본을 생성한다.

위 코드에서 아래의 배열을 복사한 부분은 d=c를 할 때 그저 c가 바라보는 값을 바라볼 뿐이다. 그리고 push는 값 d가 바라보는 값, 즉 c가 바라보는 값인 배열에다가 4를 push했기때문에 위와 같은 결과가 나타난다.

레퍼런스는 변수가 아닌 값 자체를 가리키므로 A 레퍼런스로 B레퍼런스가 가리키는 대상을 변경할 수는 없다.

var a = [1, 2, 3]
var b = a
a //[1,2,3]
b //[1,2,3]

//이후에
b = [4, 5, 6]
a //[1,2,3]
b //[4,5,6]

b가 바꼈는데 a도 바뀌는 것을 구현하려면 b가 포인터였어야했는데 자바스크립트는 포인터가 없다.

함수인자도 살펴보자

function foo(x) {
  x.push(4)
  x //[1,2,3,4]
  //그 후
  x = [4, 5, 6]
  x.push(7)
  x // [4,5,6,7]
}
var a = [1, 2, 3]
foo(a)
a //[1,2,3,4] not [4,5,6,7]

foo함수에서 매개변수로 받은 x는 처음에는 a와 같은 곳을 바라본다. 그리고 push할 때는 메모리를 하나 더 할당받아 그 곳에 푸쉬를 해놓고 아직까지 a가 바라보는 점과 x가바라보는 점은 같다.

하지만 x=[4,5,6]을 대입하는 순간 a가 참조하는 값은 그대로 [1,2,3,4]의 저장공간을 가리키는 곳을 바라보는 레퍼런스이지만 x=[4,5,6]의 저장곤간을 가리키는 곳을 바라보는 레퍼런스로 바뀌게 된다.

function foo(x) {
  x.push(4)
  x //[1,2,3,4]

  //그 후
  x.length = 0 // 기존 배열을 즉시 비운다
  x.push(4, 5, 6, 7)
  x //[4,5,6,7];
}

var a = [1, 2, 3]
foo(a)
a //[1,2,3,4]가 아닌 [4,5,6,7]

위에 x=[4,5,6] 은 새로운 공간을 생성하여 그곳을 가리키지만 x.length=0과 x.push(4,5,6,7)은 a가 가리키는 값을 바라보고 있다.

합성 값을 값-복사에 의해 효과적으로 전달하려면 손수 값의 사본을 만들어 전달한 레퍼런스가 원본을 가리키지 않게 복사하는 방식을 사용해야한다.

반대로 원시값을 레퍼런스처럼 바뀐 값을 반영시키려면 합성값으로 감싸줘야한다.

정리

  1. 자바스크립트 배열은 모든 타입의 값들을 숫자로 인덱싱한 값이다.
  2. 문자열은 유사배열이지만 불변값이므로 배열로 다루고자 할 때는 체크해야한다.
  3. 자바스크립트 숫자는 정수와 부동 소수점 숫자 모두 포함한다.
  4. null 타입은 null이란 값 하나뿐이고 undefined 타입도 값은 undefined뿐이다.
  5. undefined는 할당된 값이 없는 변수와 프로퍼티의 디폴트 값이다.
  6. NaN은 유효하지 않은 숫자의 개념이고 Number.isNan()으로 확인하자.
  7. +Infinity와 -Infinity, -0 같은 특수 값이 있고 -0은 방향을 결정할 때 꽤나 유용할 수 있다.
  8. 단순 스칼라 원시 값은 값-복사에 의해 객체나 배열등의 합성값은 레퍼런스 복사에 의해 값이 할당/전달된다.
  9. 자바스크립트의 레퍼런스는 오직 자신의 값만을 가리키므로 포인터나 주소연산자가 있는 &num들과 다름을 인지하자.
